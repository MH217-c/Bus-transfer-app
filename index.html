async function getArrival(stopId){ try { const url = `/api/bus?stationId=${stopId}`; const res = await fetch(url); const text = await res.text(); const parser = new DOMParser(); const xml = parser.parseFromString(text, "text/xml"); const items = xml.getElementsByTagName("busArrivalList"); let list = []; for (let item of items){ let route = item.getElementsByTagName("routeName")[0]?.textContent; if (!FINAL_BUSES.includes(route)) continue; let t1 = item.getElementsByTagName("predictTime1")[0]; if (t1){ list.push({ route: route, time: parseInt(t1.textContent) }); } } list.sort((a,b)=>a.time-b.time); return list.slice(0,3); } catch(e){ return []; } }
